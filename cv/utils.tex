\newcommand{\cvheadingi}[1]{
  \section*{#1}
  % \setlength{\parindent}{3ex}
}

%this version of the command above adds a space after the heading, can be used by itself 
\newcommand{\cvheading}[1]{
  \cvheadingi{#1}\vspace{1em}
}

\newenvironment{cvsection}[1]{
  % \setlength{\floatsep}{0pt}
  % \setlength{\textfloatsep}{0pt}
  % \setlength{\intextsep}{0pt}
  \cvheadingi{#1}
  \renewcommand{\arraystretch}{1.2}
  % https://tex.stackexchange.com/questions/110266/set-longtable-to-fit-the-page-width
  \begin{longtabu} to \textwidth {lX}
}{
  \end{longtabu}
  % \vspace{-1.5ex}
  \raggedbottom
}

\newenvironment{cvtext}[1]{
  % \vspace{0.15in}
  \cvheadingi{#1}
  \vspace{1em}
  \begin{justify}
  % \vspace{1.5ex}
  % \begin{minipage}{\textwidth}

  % \raggedright
}{
  \end{justify}
  % \end{tabular*}
  % \end{tabularx}
  % \end{minipage}
}
